# Allocate a random number of seats to Parties 1, 2, 3 and 4
korean_generate_random_party_seats = {
	set_variable = { korean_party1_seats = random }
	set_variable = { korean_party2_seats = random }
	set_variable = { korean_party3_seats = random }
	set_variable = { korean_party4_seats = random }

	set_temp_variable = { total = korean_party1_seats }
	add_to_temp_variable = { total = korean_party2_seats }
	add_to_temp_variable = { total = korean_party3_seats }
	add_to_temp_variable = { total = korean_party4_seats }
	divide_variable = { korean_party1_seats = total }
	divide_variable = { korean_party2_seats = total }
	divide_variable = { korean_party3_seats = total }

	multiply_variable = { korean_party1_seats = 300 }
	multiply_variable = { korean_party2_seats = 300 }
	multiply_variable = { korean_party3_seats = 300 }
	round_variable = korean_party1_seats
	round_variable = korean_party2_seats
	round_variable = korean_party3_seats

	set_variable = { korean_party4_seats = 300 }
	subtract_from_variable = { korean_party4_seats = korean_party1_seats }
	subtract_from_variable = { korean_party4_seats = korean_party2_seats }
