on_actions = {
# ROOT is capitulated country, FROM is winner
on_capitulation = {
effect = {
if = {
limit = {
ROOT  = {
original_tag = SOV
has_war_with = RCO
}
}
RCO = {
white_peace = {
        tag = SOV
        message = rco_wins_cw_peace_tt
    }
    annex_country = {
    target = SOV
    transfer_troops = no
}
}
}
}
}
}
