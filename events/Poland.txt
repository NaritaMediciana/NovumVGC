add_namespace = poland

#Invitation to Trójmorze
country_event = {
	id = poland.1
	title = poland.1.t
	desc = poland.1.d
	picture = GFX_report_event_political_deal2

	is_triggered_only = yes

	option = {	#Accept
		name = poland.1.o1
		POL = { add_to_faction = ROOT }
		set_country_flag = seeks_polish_alliance
		POL = { country_event = diplomatic_response.1 }
		ai_chance = {
			factor = 90
			modifier = {
				factor = 40
				is_neutral = yes
			}
			modifier = {
				factor = 50
				is_in_faction = yes
			}
		}
	}

	option = {	#Decline
		name = poland.1.o2
		POL = { country_event = diplomatic_response.2 }
		random_country = {
			limit = {
				tag = POL
			}
			create_wargoal = {
				type = annex_everything
				target = ROOT
			}
		}
		ai_chance = {
			factor = 10
		}
	}
}
