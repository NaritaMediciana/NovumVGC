			random_country = {
				limit = {
					is_in_faction_with = ROOT
					is_faction_leader = yes
				}
				add_opinion_modifier = {

					modifier = recent_actions_negative
				}
			}
		}
		SIA = {
			add_to_faction = ROOT
			country_event = diplomatic_response.1
		}
		
		ai_chance = {
			factor = 75
			
			modifier = {
				add = -50
				has_opinion = {
					target = SIA
					value < 10
				}
			}
			modifier = {
				add = 50
				has_opinion = {
					target = ENG
					value > 40
				}
			}
		}
	}
	
	option = {	#Ah hell nah
		name = Thailand.1.o2

		SIA = {
			add_opinion_modifier = {
				target = ROOT
				modifier = recent_actions_negative
			}
			country_event = diplomatic_response.2
		}
		
		ai_chance = {
			factor = 25
			
			modifier = {
				add = 50
				is_in_faction = yes
			}
			modifier = {
				add = 500

			}
		}
	}
