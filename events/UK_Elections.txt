		}

			popularity = 0.20
		}

		}
		add_political_power = 15
		set_country_flag = ENG_Hague_victory_2000
				add_to_variable = {
					SAMPLE_half_dial_value = 10
				}
				clamp_variable = { var = SAMPLE_half_dial_value min = 0 max = 100 }

					var = SAMPLE_half_dial_value_shown
