country_event = {
	id = Poland_Events_News.25
	title = Poland_Events_News.25.t
	desc = Poland_Events_News.25.d
	picture = POL_fascist_march_event
	
	is_triggered_only = yes 
	
	option = {
		name = Poland_Events_News.25.o1 
	    if = { #nazi victory
				limit = { 
				    AND = {
					    check_variable = { POL_popularity_of_PFN > POL_popularity_of_ONR }
						check_variable = { POL_popularity_of_PFN > POL_popularity_of_PWN }
						check_variable = { POL_popularity_of_PFN > POL_popularity_of_PRM }
					}
				}
			set_politics = {
			ruling_party = fascist
		    elections_allowed = yes
		}
		    add_popularity = { 
		    ideology = fascist
			popularity = 0.15
		}
			set_party_name = { ideology = fascist long_name = POL_fascist_party_nazi_long name = POL_fascist_party_nazi }
			if = {
					limit = {
						has_character = POL_Janusz_Bryczkowski
					}
					promote_character = POL_Janusz_Bryczkowski
				}
			}
			else_if = { #pan-slavic victory 
				limit = { 
				    AND = {
					    check_variable = { POL_popularity_of_PWN > POL_popularity_of_PRM }
						check_variable = { POL_popularity_of_PWN > POL_popularity_of_ONR }
						check_variable = { POL_popularity_of_PWN > POL_popularity_of_PFN }
					}
				}
                set_politics = {
			    ruling_party = nationalist
		        elections_allowed = yes
		 }
				add_popularity = { 
		        ideology = nationalist
			    popularity = 0.15
		    }
	    }
			else_if = { #monarchist victory
				limit = { 
				    AND = {
					    check_variable = { POL_popularity_of_PRM > POL_popularity_of_ONR }
						check_variable = { POL_popularity_of_PRM > POL_popularity_of_PWN }

			    ruling_party = monarchist
		        elections_allowed = no
		 }
				add_popularity = { 
		        ideology = monarchist
			    popularity = 0.15
		    }
			}
			else = { #Falangist victory
			set_politics = {
			ruling_party = fascist
		    elections_allowed = yes
		}
		    add_popularity = { 
		    ideology = fascist
			popularity = 0.15
		}
			}

			factor = 100
		}
