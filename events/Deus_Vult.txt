add_namespace = deus

#Finland is asked to join Sweden
country_event = {
	id = deus.1
	title = deus.1.t
	desc = deus.1.d
	picture = GFX_tenth_crusade

	is_triggered_only = yes

	option = { # Accept
		name = deus.1.a

		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		add_to_war = {
			targeted_alliance = SWE
			enemy = TUR
		}
		SWE = {
			country_event = { id = deus.2
			}
			custom_effect_tooltip = GAME_OVER_TT
		}
	}
	option = { # Refuse
		name = deus.1.b
		ai_chance = {
			factor = 0
			modifier = {
				factor = 100
				is_historical_focus_on = yes
			}
		}
		SWE = {
			country_event = { id = deus.3}
		}
	}
}

#Finland accepts the offer
